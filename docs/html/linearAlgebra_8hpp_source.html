38template <typename Vector3>
39Vector3 cross(const Vector3& vector1, const Vector3& vector2)
40{
41 assert(vector1.size() == 3 && vector2.size() == 3);
42 Vector3 result(3);
43 result[0] = vector1[1] * vector2[2] - vector1[2] * vector2[1];
44 result[1] = vector1[2] * vector2[0] - vector1[0] * vector2[2];
45 result[2] = vector1[0] * vector2[1] - vector1[1] * vector2[0];
46 return result;
47}

68template <typename Real, typename Vector>
